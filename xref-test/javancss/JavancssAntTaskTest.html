22  package javancss;
23  
24  import junit.framework.TestCase;
25  import org.apache.tools.ant.DefaultLogger;
26  import org.apache.tools.ant.Project;
27  import org.apache.tools.ant.ProjectHelper;
28  
29  import java.io.File;
30  import java.io.OutputStream;
31  import java.io.PrintStream;
32  
33  /**
34   * @author Emmanuel Bourg
35   */
36  public class JavancssAntTaskTest extends TestCase
37  {
38      private Project project;
39  
40      protected void setUp() throws Exception
41      {
42          project = new Project();
43          project.setCoreLoader( getClass().getClassLoader() );
44          project.init();
45  
46          File buildFile = new File( "target/test-classes/testbuild.xml" );
47          project.setBaseDir( buildFile.getParentFile() );
48  
49          ProjectHelper helper = ProjectHelper.getProjectHelper();
50          helper.parse( project, buildFile );
51  
52          redirectOutput( System.out );
53      }
54  
55      /**
56       * Redirects the Ant output to the specified stream.
57       */
58      private void redirectOutput( OutputStream out )
59      {
60          DefaultLogger logger = new DefaultLogger();
61          logger.setOutputPrintStream( new PrintStream( out ) );
62          logger.setMessageOutputLevel( Project.MSG_INFO );
63          project.addBuildListener( logger );
64      }
65  
66      public void testTask()
67      {
68          project.executeTarget( "run" );
69      }
70  }
