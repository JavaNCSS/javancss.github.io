24  public class ParseTest extends AbstractTestCase
25  {
26      public void testParse()
27      {
28          checkParse( 31 ); // java.net.Socket, why?
29          checkParse( 33 ); // java.text.Decimalformat, why?
30          checkParse( 34 ); // java.text.TextBoundaryData, why?
31          
32          checkParse( 48 );
33  
34          checkParse( 50 );
35  
36          checkParse( 71 ); // class declared in method
37          checkParse( 133 ); // char.class
38          checkParse( 135 ); // annotation 
39  
40          checkParse( 142 ); // JAVANCSS-12
41          checkParse( 143 ); // JAVANCSS-9
42          checkParse( 144 ); // JAVANCSS-13
43          checkParse( 145 ); // JAVANCSS-14
44          checkParse( 146 ); // JAVANCSS-17
45          checkParse( 147 ); // anonymous subcluss 
46          checkParse( 148 ); // JAVANCSS-49
47          checkParse( 149 ); // JAVANCSS-46
48          checkParse( 150 ); // JAVANCSS-53 
49          checkParse( 151 ); // JAVANCSS-45 
50          checkParse( 152 ); // JAVANCSS-57
51          checkParse( 153 ); // JAVANCSS-54
52          checkParse( 154 ); // JAVANCSS-52
53          checkParse( 155 ); // JAVANCSS-28
54          checkParse( 156 ); // hexadecimal floating-point literals
55          checkParse( 157 ); // Java 7 literals
56          checkParse( 158 ); // JAVANCSS-48
57          checkParse( 159 ); // default interface methods
58          checkParse( 160 ); // method references
59      }
60  
61      private void checkParse( int testFile )
62      {
63          Javancss pJavancss = measureTestFile( testFile );
64          assertFalse( "Parsing file Test" + testFile + ".java failed!", pJavancss.getNcss() <= 0 );
65      }
66  }
