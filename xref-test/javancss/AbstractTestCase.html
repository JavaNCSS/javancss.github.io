21  package javancss;
22  
23  import junit.framework.TestCase;
24  
25  import java.io.File;
26  
27  /**
28   * Base JavaNCSS unit-tests class.
29   *
30   * @author Herv√© Boutemy
31   * @version $Id$
32   */
33  public abstract class AbstractTestCase extends TestCase
34  {
35      private File testDir = new File( "target/test-classes/" );
36  
37      public File getTestDir()
38      {
39          return testDir;
40      }
41  
42      protected File getTestFile( String filename )
43      {
44          return new File( testDir, filename );
45      }
46  
47      protected File getTestFile( int testFileId )
48      {
49          return getTestFile( "Test" + testFileId + ".java" );
50      }
51  
52      protected Javancss measureTestFile( int testFileId )
53      {
54          return new Javancss( getTestFile( testFileId ) );
55      }
56  }
