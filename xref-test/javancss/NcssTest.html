34      private Javancss checkNcss( int testNumber, int expectedNcss )
35      {
36          Javancss javancss = measureTestFile( testNumber );
37          assertEquals( "NCSS in Test" + testNumber + ".java", expectedNcss, javancss.getNcss() );
38          return javancss;
39      }
40  
41      private void checkNcss( int testNumber )
42      {
43          Javancss javancss = measureTestFile( testNumber );
44          assertFalse( "NCSS in Test" + testNumber + ".java", javancss.getNcss() == 0 );
45      }
46  
47      private void checkNcssAndLoc( int testNumber, int expectedNcss, int expectedLoc )
48      {
49          Javancss javancss = checkNcss( testNumber, expectedNcss );
50          assertEquals( "LOC in Test" + testNumber + ".java", expectedLoc, javancss.getLOC() );
51      }
52  
53      private void checkNcssAndLoc( int testNumber, int expectedNcssAndLoc )
54      {
55          checkNcssAndLoc( testNumber, expectedNcssAndLoc, expectedNcssAndLoc );
56      }
57  
58      private Javancss checkNcssAndLoc( int testNumber )
59      {
60          Javancss javancss = measureTestFile( testNumber );
61          int ncss = javancss.getNcss();
62          int loc = javancss.getLOC();
63          assertTrue( "NCSS != LOC in Test" + testNumber + ".java. NCSS is " + ncss + ", LOC is " + loc, ncss == loc );
64          return javancss;
65      }

275     public void testNcssAndMore() throws IOException
276     {
277         final int ncss1 = 318;
278         checkNcss( 1, ncss1 );
279 
280         final int ncss6 = 565;
281         checkNcssAndLoc( 6, ncss6, 1254 );
282 
283         // Nr. 10
284         Javancss javancss = measureTestFile( 9 );
285         assertFalse( "LOC: " + javancss.getLOC(), ncss1 != javancss.getLOC() );
286 
287         checkNcssAndLoc( 10, ncss6 );
288         checkNcssAndLoc( 11 );
289 
290         javancss = checkNcssAndLoc( 12 );
291         List<FunctionMetric> vFunctions = javancss.getFunctionMetrics();
292         String sFirstFunction = vFunctions.get( 0 ).name;
293         assertNotNull( sFirstFunction );
294         /* System.out.println( sFirstFunction ); */
295         assertFalse( sFirstFunction, !sFirstFunction.equals( "Test12.readFile(URL)" ) );
296 
297         // Nr. 22
298         javancss = measureTestFile( 19 );
299         vFunctions = javancss.getFunctionMetrics();
300         sFirstFunction = vFunctions.get( 0 ).name;
301         assertFalse( sFirstFunction, !sFirstFunction.equals( "test.Test19.foo(String[],Controller)" ) );
302         sFirstFunction = vFunctions.get( 3 ).name;
303         assertFalse( !sFirstFunction.equals( "test.Test19.main(String[])" ) );
304 
305         javancss = checkNcss( 23, 10 );
306         vFunctions = javancss.getFunctionMetrics();
307         assertFalse( vFunctions.size() != 7 );
308         assertFalse( measureTestFile( 24 ).getFunctionMetrics().size() != vFunctions.size() );
309 
310         // Nr. 30
311         javancss = checkNcss( 25, 12 );
312         assertFalse( javancss.getFunctionMetrics().size() != 9 );
313 
314         javancss = measureTestFile( 56 );
315         StringWriter sw = new StringWriter();
316         javancss.printPackageNcss( sw );
317         sw.write( "\n" );
318         javancss.printObjectNcss( sw );
319         sw.write( "\n" );
320         javancss.printFunctionNcss( sw );
321 
322         String sOutput56 = sw.toString().replaceAll( "\r\n", "\n" );
323         String sCompare56 = FileUtils.readFileToString( getTestFile( "Output56.txt" ), "ISO-8859-1" );
324         assertEquals( "File test/Output56.txt and javancss output differs", sOutput56, sCompare56 );
325 
326         // check that javadocs are counted correctly
327         // after patches for additional comment counting
328         javancss = measureTestFile( 32 );
329         sw = new StringWriter();
330         javancss.printPackageNcss( sw );
331         sw.write( "\n" );
332         javancss.printObjectNcss( sw );
333         sw.write( "\n" );
334         javancss.printFunctionNcss( sw );
335 
336         String sOutput32 = sw.toString().replaceAll( "\r\n", "\n" );
337         String sCompare32 = FileUtils.readFileToString( getTestFile( "Output32.txt" ), "ISO-8859-1" );
338         assertEquals( "File test/Output32.txt and javancss output differs", sOutput32, sCompare32 );
339     }
