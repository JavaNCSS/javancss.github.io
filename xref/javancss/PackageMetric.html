21  package javancss;
22  
23  /**
24   * Basic data class to store all metrics attached to a package.
25   *
26   * @author  Chr. Clemens Lee <clemens@kclee.com>
27   * @version $Id$
28   */
29  public class PackageMetric
30      extends Metric
31  {
32      public int classes = 0;
33      public int functions = 0;
34  
35      @Override
36      public void clear()
37      {
38          super.clear();
39          classes = 0;
40          functions = 0;
41      }
42  
43      public void add( PackageMetric pPackageMetric_ )
44      {
45          if ( pPackageMetric_ == null )
46          {
47              return;
48          }
49          classes += pPackageMetric_.classes;
50          functions += pPackageMetric_.functions;
51          ncss += pPackageMetric_.ncss;
52  
53          javadocs += pPackageMetric_.javadocs;
54          javadocsLn += pPackageMetric_.javadocsLn;
55          singleLn += pPackageMetric_.singleLn;
56          multiLn += pPackageMetric_.multiLn;
57      }
58  }
